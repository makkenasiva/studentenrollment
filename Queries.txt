create table Student(
   id INT NOT NULL AUTO_INCREMENT,
   primary key (id),
   first_name VARCHAR(50),
   middle_name VARCHAR(50),
   last_name VARCHAR(50),
   parent_name VARCHAR(100),
   date_of_birth DATE,
   gender VARCHAR(10),
   mobile_number VARCHAR(15),
   email VARCHAR(100),
   address VARCHAR(200),
   grade INT,
   username VARCHAR(50)
)
